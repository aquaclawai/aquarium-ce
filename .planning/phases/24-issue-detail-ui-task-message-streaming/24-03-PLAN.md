useEffect(() => {
  if (!taskId) return;
  const wasConnected = wasConnectedRef.current;
  wasConnectedRef.current = isConnected;

**Defence-in-depth sort:** after `startTransition(() => setMessages(prev => [...prev, payload]))`, wrap the updater with a final sort step:
```tsx
startTransition(() => {
  setMessages(prev => {
    const next = [...prev, payload];
    // Cheap belt: if the last insert is already in order (hot path), skip sort.
    if (next.length < 2 || next[next.length - 1].seq > next[next.length - 2].seq) return next;
    return next.slice().sort((a, b) => a.seq - b.seq);

});
```

### B — `apps/web/src/components/issues/detail/ReconnectBanner.tsx`

```tsx
import { useEffect, useState } from 'react';
import { useTranslation } from 'react-i18next';

interface ReconnectBannerProps { isReconnecting: boolean; isReplaying: boolean }

export function ReconnectBanner({ isReconnecting, isReplaying }: ReconnectBannerProps) {
  const { t } = useTranslation();
  const [caughtUpUntil, setCaughtUpUntil] = useState(0);

  // When we transition from reconnecting/replaying → false, fade in "Caught up" for 1.5s
  useEffect(() => {
    if (!isReconnecting && !isReplaying && caughtUpUntil === 0) {
      // no transition to track yet
      return;
    }
    if (!isReconnecting && !isReplaying) {
      setCaughtUpUntil(Date.now() + 1500);
    }
  }, [isReconnecting, isReplaying, caughtUpUntil]);

  const showCaughtUp = !isReconnecting && !isReplaying && caughtUpUntil > Date.now();

  if (!isReconnecting && !isReplaying && !showCaughtUp) return null;

  const stateKey = isReconnecting ? 'reconnecting' : isReplaying ? 'replaying' : 'caught-up';
  const label = isReconnecting || isReplaying ? t('issues.detail.ws.reconnecting') : t('issues.detail.ws.replayDone');
  const cls = showCaughtUp
    ? 'bg-[var(--color-success-subtle-bg)] text-[var(--color-success-subtle-text)] opacity-0 animate-in fade-in duration-500'
    : 'bg-[var(--color-warning-subtle-bg)] text-[var(--color-warning-subtle-text)]';

  return (
    <div role="status" aria-live="polite" className={`px-3 py-1.5 text-xs rounded-md ${cls}`} data-reconnect-banner={stateKey}>
      {label}
    </div>
  );
}
```

### B — Un-skip "replay no reorder" in `tests/e2e/issue-detail.spec.ts`

Goal: during replay, drive new live broadcasts; assert final order.

Strategy: use TWO browser contexts — context A is the target observer; context B drives new broadcasts by POSTing comments / triggering task messages via an ACTOR route. But there's no public route to create task_messages — those come from the batcher. We simulate "live broadcast during replay" by racing direct DB writes against the reconnect:

```tsx
test('replay no reorder', async ({ browser, request }) => {
  const ctx = await browser.newContext();
  const page = await ctx.newPage();
  const userId = await signUpTestUser(request, uniqueEmail());
  const { issueId, taskId } = seedIssueWithRunningTask(/* ... */);

  // Seed seq 1..20
  writeDb(db => { for (let i = 1; i <= 20; i++) { /* insert */ } });

  await ctx.addCookies([{ name: 'session', value: `test:${userId}`, url: 'http://localhost:3001' }]);
  await page.goto(`http://localhost:3001/issues/${issueId}`);
  await expect(page.locator('[data-task-message-seq="20"]')).toBeVisible();

  // Kill socket
  await page.evaluate(() => (window as unknown as { __aquariumForceWsClose?: () => void }).__aquariumForceWsClose?.());

  // Immediately while disconnected: seed seq 21..30 THEN 31..40 in quick succession
  writeDb(db => { for (let i = 21; i <= 30; i++) { /* insert */ } });
  await page.waitForTimeout(1000);
  writeDb(db => { for (let i = 31; i <= 40; i++) { /* insert */ } });

  // Reconnect fires: expect 40 rows, all in order
  await expect(page.locator('[data-task-message-seq="40"]')).toBeVisible({ timeout: 15_000 });
  await expect(page.locator('[data-task-message-seq]')).toHaveCount(40);

  // Extract seq order from DOM and assert monotonic ascending
  const seqs = await page.locator('[data-task-message-seq]').evaluateAll(els => els.map(e => Number(e.getAttribute('data-task-message-seq'))));
  for (let i = 1; i < seqs.length; i++) {
    if (seqs[i] <= seqs[i - 1]) throw new Error(`out-of-order at i=${i}: ${seqs[i - 1]} >= ${seqs[i]}`);
  }
  await ctx.close();
});
```

Caveat: this test exercises the CLIENT-SIDE sort (defence in depth from Task 1) more than the SERVER-SIDE buffer — the server's replay runs sequentially under one connection. To exercise the buffer-replay-live ordering, a test would need to inject a new live broadcast DURING the server's replay phase (~1 DB query latency); that's too tight to simulate reliably from Playwright. The existing Wave 0 unit test (`ws-subscribe-task.test.ts` Test 1) IS the server-side buffer ordering proof. Playwright's role is the end-to-end "client DOM matches server DB" assertion.

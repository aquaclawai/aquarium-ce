### A — `apps/web/src/components/issues/detail/TruncationMarker.tsx`

```tsx
import { useCallback, useState } from 'react';
import { useTranslation } from 'react-i18next';
import { toast } from 'sonner';
import { api } from '../../../api';
import type { TaskMessage } from '@aquarium/shared';

const CLIENT_RENDER_CAP_BYTES = 262_144;

interface TruncationMarkerProps {
  taskId: string;
  seq: number;
  shownBytes: number;
  totalBytes: number;
  onLoad: (full: TaskMessage) => void;
  onCollapse?: () => void;
  isExpanded: boolean;
}

export function TruncationMarker({ taskId, seq, shownBytes, totalBytes, onLoad, onCollapse, isExpanded }: TruncationMarkerProps) {
  const { t } = useTranslation();
  const [loading, setLoading] = useState(false);

  const handleShowFull = useCallback(async () => {
    setLoading(true);
    try {
      const res = await api.get<TaskMessage>('/tasks/' + taskId + '/messages/' + seq + '/full');
      if (!res.ok || !res.data) throw new Error(res.error ?? 'load failed');
      onLoad(res.data);
    } catch {
      toast.error(t('issues.detail.task.showFullFailed'));
    } finally {
      setLoading(false);
    }
  }, [taskId, seq, onLoad, t]);

  return (
    <span
      className="text-xs text-muted-foreground"
      data-truncated="true"
      data-original-bytes={totalBytes}
    >
      {t('issues.detail.task.truncated', { shown: shownBytes, total: totalBytes })}
      {' '}
      {isExpanded ? (
        <button
          type="button"
          className="text-[var(--color-primary)] hover:underline"
          onClick={onCollapse}
        >
          {t('issues.detail.task.collapse')}
        </button>
      ) : (
        <button
          type="button"
          className="text-[var(--color-primary)] hover:underline disabled:opacity-50"
          onClick={handleShowFull}
          disabled={loading}
          data-action="show-full"
          data-task-id={taskId}
          data-seq={seq}
        >
          {loading ? t('issues.detail.task.showFullLoading') : t('issues.detail.task.showFull')}
        </button>
      )}
    </span>
  );
}
```

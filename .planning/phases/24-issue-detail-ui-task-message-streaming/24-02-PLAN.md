### B — `apps/web/src/components/issues/detail/useTaskStream.ts`

```tsx
import { useEffect, useRef, useState, useCallback, useDeferredValue, useTransition } from 'react';
import { api } from '../../../api';
import { useWebSocket } from '../../../context/WebSocketContext';
import type { TaskMessage, WsMessage } from '@aquarium/shared';

export interface UseTaskStreamReturn {
  messages: TaskMessage[];
  renderedMessages: TaskMessage[];
  isPaused: boolean;
  isReplaying: boolean;
  lastSeq: number;
}

export function useTaskStream({ taskId }: { taskId: string | null }): UseTaskStreamReturn {
  const { requestTaskReplay, pauseTaskStream, resumeTaskStream, addHandler, removeHandler } = useWebSocket();
  const [messages, setMessages] = useState<TaskMessage[]>([]);
  const renderedMessages = useDeferredValue(messages);
  const [isPaused, setIsPaused] = useState(false);
  const [isReplaying, setIsReplaying] = useState(false);
  const [, startTransition] = useTransition();
  const lastSeqRef = useRef(0);
  const replayingTimerRef = useRef<ReturnType<typeof setTimeout> | null>(null);

  const handleIncoming = useCallback((msg: WsMessage) => {
    if (msg.taskId !== taskId) return;
    const payload = msg.payload as TaskMessage | null | undefined;
    if (!payload || typeof payload.seq !== 'number') return;
    if (payload.seq <= lastSeqRef.current) return;        // dedup
    lastSeqRef.current = payload.seq;
    startTransition(() => { setMessages(prev => [...prev, payload]); });
    // Settle isReplaying after 500ms of no new messages
    if (replayingTimerRef.current) clearTimeout(replayingTimerRef.current);
    replayingTimerRef.current = setTimeout(() => setIsReplaying(false), 500);
  }, [taskId]);

  // Initial boot: fetch history + subscribe
  useEffect(() => {
    if (!taskId) return;
    let cancelled = false;
    (async () => {
      const res = await api.get<{ messages: TaskMessage[]; hasMore: boolean }>('/tasks/' + taskId + '/messages?afterSeq=0');
      if (cancelled) return;
      if (res.ok && res.data) {
        setMessages(res.data.messages);
        lastSeqRef.current = res.data.messages.length > 0 ? res.data.messages[res.data.messages.length - 1].seq : 0;
      }
      requestTaskReplay(taskId, lastSeqRef.current);
    })();
    addHandler('task:message', handleIncoming);
    return () => {
      cancelled = true;
      removeHandler('task:message', handleIncoming);
    };
  }, [taskId, requestTaskReplay, addHandler, removeHandler, handleIncoming]);

  // Visibility handler — pause / resume with CURRENT watermark (never 0)
  useEffect(() => {
    if (!taskId) return;
    const onVis = () => {
      if (document.hidden) {
        pauseTaskStream(taskId);
        setIsPaused(true);
      } else {
        resumeTaskStream(taskId, lastSeqRef.current);   // CURRENT watermark — NOT 0
        setIsPaused(false);
        setIsReplaying(true);
      }
    };
    document.addEventListener('visibilitychange', onVis);
    return () => document.removeEventListener('visibilitychange', onVis);
  }, [taskId, pauseTaskStream, resumeTaskStream]);

  return { messages, renderedMessages, isPaused, isReplaying, lastSeq: lastSeqRef.current };
}
```

/**
 * List tasks for a given issue, ordered by created_at DESC.
 * Workspace-scoped — cross-workspace issueIds return empty array.
 * Default limit = 20 (recent tasks for the issue; useIssueDetail only needs the latest).
 */
export async function listTasksForIssue(
  workspaceId: string,
  issueId: string,
  limit = 20,
  dbArg?: Knex,
): Promise<AgentTask[]> {
  const database = dbArg ?? (await import('../db/index.js')).db;
  const rows = await database('agent_task_queue')
    .where({ workspace_id: workspaceId, issue_id: issueId })

import { useCallback, useState } from 'react';
import { useTranslation } from 'react-i18next';
import { Card } from '../../ui/card';
import { Button } from '../../ui/button';
import { Dialog, DialogContent, DialogHeader, DialogTitle, DialogDescription, DialogFooter } from '../../ui/dialog';
import { TaskStateBadge } from './TaskStateBadge';
import { TaskMessageList } from './TaskMessageList';
import { useTaskStream } from './useTaskStream';
import { api } from '../../../api';
import { toast } from 'sonner';
import type { AgentTask } from '@aquarium/shared';

interface TaskPanelProps {
  issueId: string;
  latestTask: AgentTask | null;
  onCancel?: (taskId: string) => void;
}

export function TaskPanel({ issueId, latestTask, onCancel }: TaskPanelProps) {
  const { t } = useTranslation();
  const [confirmCancel, setConfirmCancel] = useState(false);
  const stream = useTaskStream({ taskId: latestTask?.id ?? null });
  const state = latestTask?.status ?? 'idle';
  const showCancel = state === 'running' || state === 'dispatched';

  const handleCancel = useCallback(async () => {
    if (!latestTask) return;
    const res = await api.post('/tasks/' + latestTask.id + '/cancel', {});
    if (!res.ok) { toast.error(t('issues.detail.task.cancelFailed')); return; }
    setConfirmCancel(false);
    onCancel?.(latestTask.id);
  }, [latestTask, onCancel, t]);

  return (
    <Card className="p-4" data-task-panel={latestTask?.id ?? 'idle'} data-task-state={state}>
      <div className="flex items-center justify-between mb-3">
        <div className="flex items-center gap-3">
          <h2 className="text-xl font-semibold">{t('issues.detail.task.header')}</h2>
          <TaskStateBadge state={state} />
        </div>
        {showCancel && latestTask && (
          <Button variant="outline" size="sm" onClick={() => setConfirmCancel(true)}>
            {t('issues.detail.task.cancel')}
          </Button>
        )}
      </div>
      {latestTask ? (
        <TaskMessageList taskId={latestTask.id} messages={stream.renderedMessages} isReplaying={stream.isReplaying} />
      ) : (
        <p className="text-sm text-muted-foreground">{t('issues.detail.task.idleBody')}</p>
      )}
      <Dialog open={confirmCancel} onOpenChange={setConfirmCancel}>
        <DialogContent>
          <DialogHeader>
            <DialogTitle>{t('issues.detail.task.cancelConfirm.title')}</DialogTitle>
            <DialogDescription>{t('issues.detail.task.cancelConfirm.body')}</DialogDescription>
          </DialogHeader>
          <DialogFooter>
            <Button variant="outline" onClick={() => setConfirmCancel(false)}>{t('common.buttons.cancel')}</Button>
            <Button variant="destructive" onClick={handleCancel}>{t('issues.detail.task.cancelConfirm.confirm')}</Button>
          </DialogFooter>
        </DialogContent>
      </Dialog>
    </Card>
  );
}

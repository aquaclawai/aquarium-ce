```typescript
  workspaceId: string;

  authorAgentId: string;

  trx?: Knex | Knex.Transaction;

export async function createAgentComment(args: CreateAgentCommentArgs): Promise<Comment> {
  const { workspaceId, issueId, authorAgentId, content, parentId, metadata, trx } = args;
  if (!content.trim()) throw new Error('content is required');
  const db = trx ?? (await import('../db/index.js')).db;
  // Verify issue is in workspace (IDOR guard)
  const issue = await db('issues').where({ id: issueId, workspace_id: workspaceId }).first();
  if (!issue) throw new Error('issue not found in workspace');
  // Validate parent
  if (parentId) {
    const parent = await db('comments').where({ id: parentId, issue_id: issueId }).first();
    if (!parent) throw new Error('parent comment not found in issue');
    if (parent.author_type !== 'user') throw new Error('parent comment must be a user comment');
  }
  const id = (await import('node:crypto')).randomUUID();
  const now = new Date().toISOString();
  const row = {
    id, issue_id: issueId,
    author_type: 'agent' as const,
    author_user_id: null,
    author_agent_id: authorAgentId,
    content, type: 'comment' as const,
    parent_id: parentId ?? null,
    metadata: adapter.jsonValue(metadata ?? {}),
    created_at: now, updated_at: now,
  };
  await db('comments').insert(row);
  const inserted = await db('comments').where({ id }).first();
  return toComment(inserted);
}
```

```tsx
import { useCallback, useState, useRef, useEffect } from 'react';
import { useTranslation } from 'react-i18next';
import { toast } from 'sonner';
import { Card } from '../../ui/card';
import { Button } from '../../ui/button';
import type { Issue, Comment, AgentTask } from '@aquarium/shared';

const MAX_CHARS = 8000;

interface ChatComposerProps {
  issue: Issue;
  lastUserCommentId: string | null;
  onSubmit: (args: { content: string; triggerCommentId: string | null }) => Promise<{ comment: Comment; enqueuedTask: AgentTask | null }>;
  disabled?: boolean;
}

export function ChatComposer({ issue, lastUserCommentId, onSubmit, disabled }: ChatComposerProps) {
  const { t } = useTranslation();
  const [content, setContent] = useState('');
  const [submitting, setSubmitting] = useState(false);
  const textareaRef = useRef<HTMLTextAreaElement>(null);
  const hasAssignee = issue.assigneeId !== null;
  const effectiveDisabled = disabled || !hasAssignee;
  const showCharCounter = content.length > MAX_CHARS - 200;

  const handleSend = useCallback(async () => {
    const trimmed = content.trim();
    if (!trimmed || effectiveDisabled || submitting) return;
    if (!hasAssignee) { toast.error(t('chat.composer.noAssignee')); return; }
    setSubmitting(true);
    try {
      await onSubmit({ content: trimmed.slice(0, MAX_CHARS), triggerCommentId: lastUserCommentId });
      setContent('');
    } catch {
      toast.error(t('chat.composer.sendFailed'));
    } finally {
      setSubmitting(false);
    }
  }, [content, effectiveDisabled, submitting, hasAssignee, lastUserCommentId, onSubmit, t]);

  const handleKeyDown = useCallback((e: React.KeyboardEvent<HTMLTextAreaElement>) => {
    if ((e.metaKey || e.ctrlKey) && e.key === 'Enter') {
      e.preventDefault();
      void handleSend();
    }
  }, [handleSend]);

  return (
    <div className="sticky bottom-0 bg-background pt-4" style={{ zIndex: 'var(--z-sticky)' }}>
```

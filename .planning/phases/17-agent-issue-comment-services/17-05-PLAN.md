async function pollUntil<T>(fn: () => Promise<T | null> | T | null, timeoutMs: number, intervalMs = 100, label = 'condition'): Promise<T> {
  const deadline = Date.now() + timeoutMs;
  let last: T | null = null;
  while (Date.now() < deadline) {
    last = await fn();
    if (last !== null) return last;
    await new Promise((r) => setTimeout(r, intervalMs));
  }
  throw new Error(`[issues-agents-comments.spec] pollUntil '${label}' timed out after ${timeoutMs}ms (last=${String(last)})`);
}

test('ISSUE-01: CRUD + issue_number monotonicity under concurrency', async ({ request }) => {
  // Baseline counter before burst
  const beforeCounter = readDb((db) => {
    const row = db.prepare(`SELECT issue_counter FROM workspaces WHERE id = 'AQ'`).get() as { issue_counter: number };

  });
  expect(createRes.status()).toBe(201);

  expect(firstIssue.issueNumber).toBe(beforeCounter + 1);
  issueId = firstIssue.id;

  // Concurrent burst of 5 — all must get unique sequential issue_numbers
  const burstResults = await Promise.all(
    Array.from({ length: 5 }).map(() =>
      request.post(`${API}/issues`, { data: { title: uniqueName('burst') } }).then((r) => r.json()),
    ),
  );
  const burstNumbers = burstResults.map((b: { data: IssueShape }) => b.data.issueNumber).sort((a, b) => a - b);
  expect(burstNumbers.length).toBe(5);
  expect(new Set(burstNumbers).size).toBe(5); // all unique
  // Max - min = 4 if perfectly sequential (no gaps). Allow 1-gap tolerance for fairness.
  expect(burstNumbers[burstNumbers.length - 1] - burstNumbers[0]).toBeLessThanOrEqual(4);

    data: { priority: 'high', description: 'updated description' },

  expect(updateRes.status()).toBe(200);

test('ISSUE-05: reorder midpoint + renumber sweep on precision collapse', async ({ request }) => {
  // Create 3 issues, reorder them to positions 1.0 and 1.0000005 manually (forcing collapse)
  const makeIssue = async (title: string) =>
    (await (await request.post(`${API}/issues`, { data: { title } })).json()).data as IssueShape;
  const a = await makeIssue(uniqueName('reorder-a'));
  const b = await makeIssue(uniqueName('reorder-b'));
  const c = await makeIssue(uniqueName('reorder-c'));

  // Direct-SQL inject collapsing positions on a and b
  writeDb((db) => {
    db.prepare(`UPDATE issues SET position = 1.0 WHERE id = ?`).run(a.id);
    db.prepare(`UPDATE issues SET position = 1.0000005 WHERE id = ?`).run(b.id);
  });

  // Reorder c between them → must trigger workspace renumber (|a-b| < 1e-6)
  const reorderRes = await request.post(`${API}/issues/${c.id}/reorder`, {
    data: { beforeId: a.id, afterId: b.id },
  });
  expect(reorderRes.status()).toBe(200);

  // Verify renumber swept to multiples of RENUMBER_STEP=1000
  const positions = readDb((db) =>
    db.prepare(`SELECT id, position FROM issues WHERE workspace_id='AQ' AND position IS NOT NULL ORDER BY position ASC`).all() as Array<{ id: string; position: number }>,
  );
  const gaps = positions.slice(1).map((p, i) => p.position - positions[i].position);
  // After renumber sweep, every gap should be >= RENUMBER_STEP/2 (the midpoint re-insertion leaves at least step/2 between neighbours)
  expect(gaps.every((g) => g >= 500)).toBeTruthy();
  // c must be between a and b in the final ordering (neighbouring ids)
  const idOrder = positions.map((p) => p.id);
  const aIdx = idOrder.indexOf(a.id), bIdx = idOrder.indexOf(b.id), cIdx = idOrder.indexOf(c.id);
  expect(cIdx > aIdx && cIdx < bIdx).toBeTruthy();
});

    STEP 1 — Create `apps/server/src/task-dispatch/hosted-orphan-sweep.ts` (model after `task-reaper.ts` reapOnce, but ONE-SHOT — no setInterval, no start/stop):

    ```typescript
    import type { Knex } from 'knex';
    import { db as defaultDb } from '../db/index.js';
    import { broadcast } from '../ws/index.js';

    /**
     * Boot-time orphan sweep for hosted-instance tasks — Phase 20, HOSTED-04.
     *
     * Scope:
     *   • SELECT agent_task_queue rows WHERE status IN ('dispatched','running')
     *     AND runtime.kind='hosted_instance' (JOIN on runtimes).
     *   • UPDATE those rows to status='failed', error='hosted-orphan-on-boot',
     *     completed_at=now, updated_at=now (with ST6 race guard re-applying the
     *     status filter so a concurrent legitimate transition is not clobbered).
     *   • Fire task:failed WS broadcasts for EVERY row returned by the SELECT
     *     (see "ST6 benign over-broadcast" note below).
     *
     * Ordering (server-core.ts):
     *   Step 9a runtime-bridge reconcile  — EXISTING
     *   Step 9b hosted orphan sweep       — THIS FILE (before task-reaper)
     *   Step 9c task-reaper               — EXISTING
     *   Step 9d hosted worker             — Plan 20-02 (after task-reaper)
     *   Step 9e offline sweeper           — EXISTING
     *
     * Why BEFORE 9c: task-reaper's generic "Reaper: dispatched > 5 min" error
     * would overwrite the correct 'hosted-orphan-on-boot' reason HOSTED-04 requires
     * if 9c ran first.
     *
     * ST6 race guard (matches task-reaper.ts:84/96): the UPDATE re-applies
     * `whereIn('status', ['dispatched','running'])` so a concurrent legitimate
     * transition (daemon completing a task in the 1ms between SELECT and UPDATE)
     * cannot be clobbered. Under server-core Step 9b this race is effectively
     * impossible (HTTP not yet listening), but the guard is cheap.
     *
     * ST6 benign over-broadcast: if a row is returned by the SELECT but its
     * status transitioned between SELECT and UPDATE (so the UPDATE guard skipped
     * it), we still emit a `task:failed` broadcast for it. This is acceptable
     * because at boot time NO WS CLIENTS ARE CONNECTED (server is still in the
     * startup sequence before HTTP.listen). The extra event is dropped by the
     * empty client map in broadcast(). A more correct alternative would be to
     * re-SELECT after UPDATE (or use RETURNING if the Knex SQLite driver
     * supports it), but the extra round trip isn't worth the complexity for a
     * benign failure mode.
     *
     * Daemon tasks are NOT touched: the join filter `r.kind='hosted_instance'`
     * excludes local_daemon and external_cloud_daemon rows.
     */
    export async function failOrphanedHostedTasks(
      dbOverride?: Knex,
    ): Promise<{ failed: number; rows: Array<{ taskId: string; issueId: string; workspaceId: string }> }> {
      const kx = dbOverride ?? defaultDb;
      const nowIso = new Date().toISOString();

      // SELECT candidate rows first so we can broadcast per-row after UPDATE.
      const rows = (await kx('agent_task_queue as q')
        .join('runtimes as r', 'r.id', 'q.runtime_id')
        .where('r.kind', 'hosted_instance')
        .whereIn('q.status', ['dispatched', 'running'])
        .select('q.id', 'q.issue_id', 'q.workspace_id', 'q.status')) as Array<{
          id: string; issue_id: string; workspace_id: string; status: 'dispatched' | 'running';
        }>;

      if (rows.length === 0) return { failed: 0, rows: [] };

      const ids = rows.map((r) => r.id);
      // ST6 race guard (matches task-reaper.ts pattern).
      const failed = await kx('agent_task_queue')
        .whereIn('id', ids)
        .whereIn('status', ['dispatched', 'running'])
        .update({
          status: 'failed',
          error: 'hosted-orphan-on-boot',
          completed_at: nowIso,
          updated_at: nowIso,
        });

      // Broadcast AFTER commit — single-statement UPDATE autocommits before
      // returning (SQ5: no I/O in txn).
      //
      // ST6 benign over-broadcast accepted: at boot time no WS clients are
      // connected, so an extra task:failed event for a concurrently-completed
      // row is harmless (broadcast() iterates an empty client map). Filtering
      // by array index would be INCORRECT — it filters by position, not by
      // actual updated-row identity. Re-SELECTing after UPDATE or using
      // RETURNING would be more correct but adds a round trip; the benign
      // failure mode at boot doesn't justify the extra complexity.
      const reapedRows = rows.map((r) => ({
        taskId: r.id,
        issueId: r.issue_id,
        workspaceId: r.workspace_id,
      }));

      for (const r of reapedRows) {
        activeBroadcast(r.workspaceId, {
          type: 'task:failed',
          taskId: r.taskId,
          issueId: r.issueId,
          payload: { taskId: r.taskId, issueId: r.issueId },
        });
      }

      if (failed > 0) {
        console.log(`[hosted-orphan-sweep] failed ${failed} hosted-orphan task(s) on boot`);
      }

      return { failed, rows: reapedRows };
    }

    // Test hook: swap the broadcast function so unit tests can count invocations
    // without standing up a real WS server.
    let activeBroadcast: typeof broadcast = broadcast;
    export function __setBroadcastForTests__(fn: typeof broadcast): void { activeBroadcast = fn; }
    export function __resetBroadcastForTests__(): void { activeBroadcast = broadcast; }
    ```

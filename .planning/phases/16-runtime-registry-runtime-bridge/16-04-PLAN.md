import { test, expect } from '@playwright/test';
import type { Page } from '@playwright/test';
import Database from 'better-sqlite3';
import { homedir } from 'node:os';
import { join } from 'node:path';

const BASE = 'http://localhost:5173';
const API = 'http://localhost:3001/api';
const DB_PATH = process.env.AQUARIUM_DB_PATH || join(homedir(), '.aquarium', 'aquarium.db');
const DEFAULT_WORKSPACE_ID = 'AQ';

function uniqueEmail(): string {
  return `runtimes-${Date.now()}-${Math.random().toString(36).slice(2, 8)}@e2e.test`;
}

function uniqueName(prefix: string): string {
  return `${prefix}-${Date.now()}-${Math.random().toString(36).slice(2, 6)}`;
}

async function pollUntil<T>(
  fn: () => Promise<T | null>,
  timeoutMs: number,
  intervalMs = 100,
  label = 'condition',
): Promise<T> {
  const deadline = Date.now() + timeoutMs;
  let last: T | null = null;
  while (Date.now() < deadline) {
    last = await fn();
    if (last !== null) return last;
    await new Promise((r) => setTimeout(r, intervalMs));
  }
  throw new Error(`[runtimes.spec] pollUntil '${label}' timed out after ${timeoutMs}ms`);
}

interface RuntimeShape {
  id: string;
  name: string;
  kind: 'local_daemon' | 'external_cloud_daemon' | 'hosted_instance';
  provider: string;
  status: 'online' | 'offline' | 'error';
  instanceId: string | null;
  daemonId: string | null;
  lastHeartbeatAt: string | null;
}

async function listRuntimes(page: Page): Promise<RuntimeShape[]> {
  const res = await page.request.get(`${API}/runtimes`);
  expect(res.ok()).toBeTruthy();
  const body = (await res.json()) as { ok: boolean; data?: RuntimeShape[]; error?: string };
  expect(body.ok).toBe(true);
  return body.data ?? [];
}

async function findMirrorByInstance(page: Page, instanceId: string): Promise<RuntimeShape | null> {
  const runtimes = await listRuntimes(page);
  return runtimes.find((r) => r.instanceId === instanceId) ?? null;
}

  const email = uniqueEmail();
  const password = 'RuntimesTest123!';
  const displayName = 'Runtimes Tester';
  const instanceName = uniqueName('rt-inst');
  let instanceId = '';

    // This catches "wrong DB_PATH" misconfiguration before the first SQL query.

    probe.close();
  });

  test('signup as new user', async ({ page }) => {

    await page.fill('#displayName', displayName);
    await page.fill('#email', email);
    await page.fill('#password', password);
    await page.click('button[type="submit"]');
    await expect(page.locator('.app-layout')).toBeVisible({ timeout: 10000 });
  });

  test('RT-01: GET /api/runtimes returns 200 with a Runtime[] shape', async ({ page }) => {
    // Unauthenticated request must be rejected
    const anonContext = await page.context().browser()!.newContext();
    const anonRes = await anonContext.request.get(`${API}/runtimes`);
    expect(anonRes.status()).toBe(401);
    await anonContext.close();

    // Authenticated request returns 200 with { ok, data: [] }

    expect(Array.isArray(runtimes)).toBe(true);
    // On a fresh DB post-signup with no instances yet, list is empty OR contains

    for (const r of runtimes) {

      expect(['online', 'offline', 'error']).toContain(r.status);
    }
  });

  test('RT-02: creating an instance produces a mirror runtime within 2s', async ({ page }) => {
    const createRes = await page.request.post(`${API}/instances`, {
      data: { name: instanceName, agentType: 'openclaw' },
    });
    expect(createRes.ok()).toBeTruthy();
    const createBody = (await createRes.json()) as { ok: boolean; data?: { id: string } };

      2000,
      100,
      `mirror runtime for instance ${instanceId}`,

    const elapsed = Date.now() - t0;

    expect(mirror.kind).toBe('hosted_instance');
    expect(mirror.provider).toBe('hosted');
    expect(mirror.name).toBe(instanceName);
    expect(mirror.daemonId).toBeNull();
    console.log(`[RT-02] mirror appeared in ${elapsed}ms`);
  });

  test('RT-03: renaming instance propagates to mirror.name within 2s', async ({ page }) => {
    const renamed = `${instanceName}-renamed`;
    const patchRes = await page.request.patch(`${API}/instances/${instanceId}/config`, {
      data: { agentName: renamed },
    });
    expect(patchRes.ok()).toBeTruthy();

    const t0 = Date.now();
    const mirror = await pollUntil(
      async () => {
        const m = await findMirrorByInstance(page, instanceId);
        return m && m.name === renamed ? m : null;
      },
      2000,
      100,
      `mirror.name = ${renamed}`,

    const elapsed = Date.now() - t0;
    expect(mirror.name).toBe(renamed);
    console.log(`[RT-03 rename] mirror.name updated in ${elapsed}ms`);
  });

  test('RT-03: deleting instance removes mirror runtime within 2s (FK CASCADE)', async ({ page }) => {
    const delRes = await page.request.delete(`${API}/instances/${instanceId}?purge=true`);
    expect(delRes.ok()).toBeTruthy();

    const t0 = Date.now();
    await pollUntil(
      async () => {
        const m = await findMirrorByInstance(page, instanceId);
        return m === null ? 'deleted' : null;
      },
      2000,
      100,
      `mirror row for ${instanceId} gone`,

    const elapsed = Date.now() - t0;
    console.log(`[RT-03 delete] mirror CASCADE removed in ${elapsed}ms`);
  });

  test('RT-04: derived status + ST1 — stored r.status never written for hosted rows', async ({ page }) => {
    // Create a second instance specifically for RT-04
    const rt4Name = uniqueName('rt04');
    const createRes = await page.request.post(`${API}/instances`, {
      data: { name: rt4Name, agentType: 'openclaw' },
    });
    expect(createRes.ok()).toBeTruthy();
    const createBody = (await createRes.json()) as { ok: boolean; data?: { id: string; status: string } };
    const rt4InstanceId = createBody.data?.id ?? '';
    expect(rt4InstanceId).toBeTruthy();

    await pollUntil(() => findMirrorByInstance(page, rt4InstanceId), 2000, 100, 'rt04 mirror');

    // Direct SQL: the stored runtimes.status column for a hosted_instance row
    // must be the 'offline' placeholder (ST1 HARD: bridge never writes this column
    // for hosted rows post-INSERT).
    const dbRead = new Database(DB_PATH, { readonly: true });
    const storedRow = dbRead
      .prepare(
        `SELECT r.status AS stored_status, i.status AS instance_status
         FROM runtimes r
         JOIN instances i ON r.instance_id = i.id
         WHERE r.instance_id = ? AND r.workspace_id = ?`,
      )
      .get(rt4InstanceId, DEFAULT_WORKSPACE_ID) as
      | { stored_status: string; instance_status: string }
      | undefined;
    dbRead.close();

    expect(storedRow).toBeDefined();
    expect(storedRow!.stored_status).toBe('offline'); // ST1: placeholder never mutated
    expect(['created', 'stopped']).toContain(storedRow!.instance_status); // instance never started
    // GET-side: API returns derived status from the CASE WHEN (stopped → offline, running → online, error → error)
    const mirror = await findMirrorByInstance(page, rt4InstanceId);
    expect(mirror).toBeTruthy();
    expect(mirror!.status).toBe('offline'); // derived status tracks instance.status

    // Cleanup
    await page.request.delete(`${API}/instances/${rt4InstanceId}?purge=true`);
  });

  test('RT-05: daemon runtime with stale heartbeat flips offline within one sweep tick', async ({ page }) => {
    // Fast path: inject a fake daemon runtime directly into the DB. The offline sweeper
    // runs every 30s and transitions rows with last_heartbeat_at < (now - 90s) to offline.
    // We start with status='online' and last_heartbeat_at=now-120s, so after ≤ 45s
    // (one full 30s cycle + generous margin for first-tick jitter), status must be 'offline'.
    const dbWrite = new Database(DB_PATH);
    const fakeId = `daemon-test-${Date.now()}`;
    const staleIso = new Date(Date.now() - 120_000).toISOString();

    dbWrite
      .prepare(
        `INSERT INTO runtimes
           (id, workspace_id, name, kind, provider, status, daemon_id, instance_id,
            metadata, last_heartbeat_at, created_at, updated_at)
         VALUES (?, ?, ?, 'local_daemon', 'claude', 'online', ?, NULL,
            '{}', ?, datetime('now'), datetime('now'))`,
      )
      .run(fakeId, DEFAULT_WORKSPACE_ID, 'RT-05 test daemon', `daemon-${fakeId}`, staleIso);

    // Seed check: we inserted a row with status='online'
    const pre = dbWrite
      .prepare(`SELECT status FROM runtimes WHERE id = ?`)
      .get(fakeId) as { status: string } | undefined;
    expect(pre?.status).toBe('online');

    const t0 = Date.now();
    // Wait up to 45s for the sweeper to flip it. In CI, retry at 500ms intervals.
    await pollUntil(
      () => {
        const row = dbWrite.prepare(`SELECT status FROM runtimes WHERE id = ?`).get(fakeId) as
          | { status: string }
          | undefined;
        return row?.status === 'offline' ? 'flipped' : null;
      },
      45_000,
      500,
      `daemon ${fakeId} status -> offline`,
    );
    const elapsed = Date.now() - t0;
    console.log(`[RT-05] sweeper flipped daemon to offline in ${elapsed}ms`);

    // Cleanup the fake row so other tests aren't affected
    dbWrite.prepare(`DELETE FROM runtimes WHERE id = ?`).run(fakeId);
    dbWrite.close();
  });

  test('ST1 global proof: r.status for every hosted_instance row is still the offline placeholder', async () => {
    const db = new Database(DB_PATH, { readonly: true });
    const rows = db
      .prepare(
        `SELECT r.id, r.status, i.status AS instance_status
         FROM runtimes r
         LEFT JOIN instances i ON r.instance_id = i.id
         WHERE r.kind = 'hosted_instance' AND r.workspace_id = ?`,
      )
      .all(DEFAULT_WORKSPACE_ID) as Array<{ id: string; status: string; instance_status: string | null }>;
    db.close();

    for (const row of rows) {
      expect(row.status, `hosted mirror ${row.id} r.status must be offline placeholder (ST1)`).toBe('offline');
    }
  });
});

Ship the 9 daemon-auth REST endpoints plus the rate-limiter topology change. Every endpoint is a thin HTTP wrapper over Phase 16/17/18 service functions — no new business logic. The rate-limiter change is two `skip:` predicates on existing `/api/` limiters + one new `daemonBucket` limiter keyed by token hash.

**Purpose:** Daemons can now register, heartbeat, claim tasks, report progress/messages/completion, and poll status — all under workspace-scoped authentication. This unblocks Phase 20 (hosted driver) and Phase 21 (daemon CLI + Claude backend).

**Scope locked here** (per 19-01 objective):
- Q1: `/register` rejects body `workspaceId` ≠ `req.daemonAuth.workspaceId` with 400 (defence-in-depth).
- Q2: `skip:` predicate via `req.originalUrl.startsWith('/api/daemon/')`.
- Q4: `/progress` emits WS event ONLY — no DB write (Phase 15 schema has no `progress` column).
- Q6: 1000 requests / 60s per-token bucket.
- Q8: `/heartbeat` returns 409 when `daemonId` null.
- Workspace scoping: every service call is filtered by `req.daemonAuth.workspaceId` (cross-workspace runtime id → 404 via `runtime-registry.getById`).

    **Step B — create apps/server/src/routes/daemon.ts** (~280 lines). Structure:

    ```typescript
    import { Router } from 'express';
    import rateLimit from 'express-rate-limit';
    import { requireDaemonAuth } from '../middleware/daemon-auth.js';
    import {
      upsertDaemonRuntime,
      updateHeartbeat,
      setRuntimeOffline,
      getById as getRuntimeById,
    } from '../services/runtime-registry.js';
    import {
      claimTask,
      startTask,
      completeTask,
      failTask,
      isTaskCancelled,
    } from '../services/task-queue-store.js';
    import { appendTaskMessage, type PendingTaskMessage } from '../task-dispatch/task-message-batcher.js';
    import { broadcast } from '../ws/index.js';
    import { db } from '../db/index.js';
    import type {
      ApiResponse,
      Runtime,
      DaemonRegisterRequest,
      DaemonRegisterResponse,
      ClaimedTask,
      TaskStatus,
      TerminalResult,
      TaskMessageType,
    } from '@aquarium/shared';

    // Note: TerminalResult lives in task-queue-store.ts not shared — re-type here or export from there.
    // Use `import type { TerminalResult } from '../services/task-queue-store.js';` — it's already exported.

    const router = Router();
    router.use(requireDaemonAuth);

    // §DAEMON-08 per-token bucket. Mounted AFTER requireDaemonAuth so req.daemonAuth.tokenHash is populated.
    const daemonBucket = rateLimit({
      windowMs: 60 * 1000,
      limit: 1000,
      standardHeaders: true,
      legacyHeaders: false,
      validate: { trustProxy: false, xForwardedForHeader: false },
      keyGenerator: (req) => {
        // Guarded — middleware above already populated this. Fall back to IP in the impossible case.
        return req.daemonAuth?.tokenHash ?? req.ip ?? 'unknown';
      },
    });
    if (process.env.NODE_ENV === 'production') {
      router.use(daemonBucket);
    }
    // Dev/test parity with existing server-core gate (limiter disabled in dev for E2E).

    // ── POST /register — DAEMON-01 ──────────────────────────────────────────
    router.post('/register', async (req, res) => {
      try {
        const body = req.body as Partial<DaemonRegisterRequest>;
        if (!body || typeof body !== 'object') {
          res.status(400).json({ ok: false, error: 'invalid body' } satisfies ApiResponse);
          return;
        }
        if (!body.daemonId || typeof body.daemonId !== 'string') {
          res.status(400).json({ ok: false, error: 'daemonId required' } satisfies ApiResponse);
          return;
        }
        if (typeof body.workspaceId === 'string' && body.workspaceId !== req.daemonAuth!.workspaceId) {
          res.status(400).json({ ok: false, error: 'workspace mismatch' } satisfies ApiResponse);
          return;
        }
        if (!Array.isArray(body.runtimes) || body.runtimes.length === 0) {
          res.status(400).json({ ok: false, error: 'runtimes array required' } satisfies ApiResponse);
          return;
        }

        // Persist daemon_id on the token row so subsequent /heartbeat can enforce lifecycle.
        await db('daemon_tokens')
          .where({ id: req.daemonAuth!.tokenId })
          .update({ daemon_id: body.daemonId, updated_at: new Date().toISOString() });

        const deviceInfo = body.deviceName
          ? { deviceName: body.deviceName, cliVersion: body.cliVersion ?? '', launchedBy: body.launchedBy ?? '' }
          : null;

        const created: Runtime[] = [];
        for (const rt of body.runtimes) {
          const id = await upsertDaemonRuntime({
            workspaceId: req.daemonAuth!.workspaceId,
            daemonId: body.daemonId,
            provider: rt.provider,
            name: rt.name,
            deviceInfo,
            ownerUserId: null,
            kind: 'local_daemon',
          });
          const runtime = await getRuntimeById(req.daemonAuth!.workspaceId, id);
          if (runtime) created.push(runtime);
        }

        const payload: DaemonRegisterResponse = { runtimes: created };
        res.json({ ok: true, data: payload } satisfies ApiResponse<DaemonRegisterResponse>);
      } catch (err) {
        const msg = err instanceof Error ? err.message : String(err);
        res.status(500).json({ ok: false, error: msg } satisfies ApiResponse);
      }
    });

    // ── POST /heartbeat — DAEMON-02 ─────────────────────────────────────────
    router.post('/heartbeat', async (req, res) => {
      try {
        if (!req.daemonAuth!.daemonId) {
          res.status(409).json({ ok: false, error: 'daemon not registered — call /register first' } satisfies ApiResponse);
          return;
        }
        const runtimeIds = Array.isArray(req.body?.runtimeIds) ? (req.body.runtimeIds as string[]) : [];
        for (const id of runtimeIds) {
          if (typeof id !== 'string') continue;
          const runtime = await getRuntimeById(req.daemonAuth!.workspaceId, id);
          if (!runtime) continue; // silently skip foreign IDs; daemon's cache may be stale
          await updateHeartbeat(id);
        }
        res.json({ ok: true, data: { pendingPings: [], pendingUpdates: [] } } satisfies ApiResponse<{ pendingPings: unknown[]; pendingUpdates: unknown[] }>);
      } catch (err) {
        const msg = err instanceof Error ? err.message : String(err);
        res.status(500).json({ ok: false, error: msg } satisfies ApiResponse);
      }
    });

    // ── POST /deregister — DAEMON-03 ────────────────────────────────────────
    router.post('/deregister', async (req, res) => {
      try {
        const runtimeIds = Array.isArray(req.body?.runtimeIds) ? (req.body.runtimeIds as string[]) : [];
        for (const id of runtimeIds) {
          if (typeof id !== 'string') continue;
          const runtime = await getRuntimeById(req.daemonAuth!.workspaceId, id);
          if (!runtime) continue;
          await setRuntimeOffline(id);
        }
        res.json({ ok: true, data: { ok: true } } satisfies ApiResponse<{ ok: boolean }>);
      } catch (err) {
        const msg = err instanceof Error ? err.message : String(err);
        res.status(500).json({ ok: false, error: msg } satisfies ApiResponse);
      }
    });

    // ── POST /runtimes/:id/tasks/claim — DAEMON-04 ──────────────────────────
    router.post('/runtimes/:id/tasks/claim', async (req, res) => {
      try {
        const runtime = await getRuntimeById(req.daemonAuth!.workspaceId, req.params.id);
        if (!runtime) {
          res.status(404).json({ ok: false, error: 'runtime not found' } satisfies ApiResponse);
          return;
        }
        const task = await claimTask(req.params.id);
        res.json({ ok: true, data: { task } } satisfies ApiResponse<{ task: ClaimedTask | null }>);
      } catch (err) {
        const msg = err instanceof Error ? err.message : String(err);
        res.status(500).json({ ok: false, error: msg } satisfies ApiResponse);
      }
    });

    // ── POST /tasks/:id/start — DAEMON-05a ──────────────────────────────────
    router.post('/tasks/:id/start', async (req, res) => {
      try {
        const result = await startTask(req.params.id);
        res.json({ ok: true, data: result } satisfies ApiResponse<{ started: boolean; status: TaskStatus }>);
      } catch (err) {
        const msg = err instanceof Error ? err.message : String(err);
        res.status(500).json({ ok: false, error: msg } satisfies ApiResponse);
      }
    });

    // ── POST /tasks/:id/progress — DAEMON-05b (WS-only per Q4) ──────────────
    router.post('/tasks/:id/progress', async (req, res) => {
      try {
        const progress = typeof req.body?.progress === 'number' ? req.body.progress : null;
        const note = typeof req.body?.note === 'string' ? req.body.note : undefined;
        // No DB write — progress not part of Phase 15 schema. WS fan-out only.
        const row = await db('agent_task_queue')
          .where({ id: req.params.id, workspace_id: req.daemonAuth!.workspaceId })
          .first('issue_id');
        if (!row) {
          res.status(404).json({ ok: false, error: 'task not found' } satisfies ApiResponse);
          return;
        }
        broadcast(req.daemonAuth!.workspaceId, {
          type: 'task:progress',
          taskId: req.params.id,
          issueId: row.issue_id as string,
          payload: { taskId: req.params.id, issueId: row.issue_id as string, progress, note },
        });
        res.json({ ok: true, data: { ok: true } } satisfies ApiResponse<{ ok: boolean }>);
      } catch (err) {
        const msg = err instanceof Error ? err.message : String(err);
        res.status(500).json({ ok: false, error: msg } satisfies ApiResponse);
      }
    });

    // ── POST /tasks/:id/messages — DAEMON-05c (batched ingest) ──────────────
    const MAX_BATCH = 100;
    const MAX_BATCH_BYTES = 64 * 1024;
    router.post('/tasks/:id/messages', async (req, res) => {
      try {
        const messages = Array.isArray(req.body?.messages) ? (req.body.messages as unknown[]) : null;
        if (!messages) {
          res.status(400).json({ ok: false, error: 'messages array required' } satisfies ApiResponse);
          return;
        }
        if (messages.length > MAX_BATCH) {
          res.status(413).json({ ok: false, error: 'batch too large (max 100)' } satisfies ApiResponse);
          return;
        }
        if (JSON.stringify(req.body).length > MAX_BATCH_BYTES) {
          res.status(413).json({ ok: false, error: 'batch too large (>64KB)' } satisfies ApiResponse);
          return;
        }

        // Resolve issueId once via DB (cheaper than per-message DB read).
        const task = await db('agent_task_queue')
          .where({ id: req.params.id, workspace_id: req.daemonAuth!.workspaceId })
          .first('issue_id');
        if (!task) {
          res.status(404).json({ ok: false, error: 'task not found' } satisfies ApiResponse);
          return;
        }

        let accepted = 0;
        for (const raw of messages) {
          if (!raw || typeof raw !== 'object') continue;
          const m = raw as Record<string, unknown>;
          const type = m.type as TaskMessageType | undefined;
          if (!type) continue;
          const pending: PendingTaskMessage = {
            type,
            tool: typeof m.tool === 'string' ? m.tool : null,
            content: typeof m.content === 'string' ? m.content : null,
            input: m.input,
            output: m.output,
            metadata: (m.metadata && typeof m.metadata === 'object') ? (m.metadata as Record<string, unknown>) : {},
            workspaceId: req.daemonAuth!.workspaceId,
            issueId: task.issue_id as string,
          };
          appendTaskMessage(req.params.id, pending);
          accepted++;
        }
        res.json({ ok: true, data: { accepted } } satisfies ApiResponse<{ accepted: number }>);
      } catch (err) {
        const msg = err instanceof Error ? err.message : String(err);
        res.status(500).json({ ok: false, error: msg } satisfies ApiResponse);
      }
    });

    // ── POST /tasks/:id/complete — DAEMON-05d ───────────────────────────────
    router.post('/tasks/:id/complete', async (req, res) => {
      try {
        const result = await completeTask(req.params.id, req.body?.result);
        // ALWAYS 200 — discarded: true is not an error (TASK-06).
        res.json({ ok: true, data: result } satisfies ApiResponse<TerminalResult>);
      } catch (err) {
    ```

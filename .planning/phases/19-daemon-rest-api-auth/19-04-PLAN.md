    **Step A — write tests/e2e/fixtures/daemon-helpers.ts**:

    ```typescript
    import type { APIRequestContext } from '@playwright/test';

    /**
     * Creates a daemon token via POST /api/daemon-tokens (user-auth).
     * Returns the id + plaintext for subsequent daemon calls.
     * Caller must ensure `request` has a logged-in cookie session (via UI login
     * or /api/auth/test-signup test cookie).
     */
    export async function mintDaemonToken(
      request: APIRequestContext,
      name: string,
    ): Promise<{ id: string; plaintext: string }> {
      const res = await request.post('/api/daemon-tokens', { data: { name } });
      if (res.status() !== 200) {
        throw new Error(`mintDaemonToken: expected 200, got ${res.status()}: ${await res.text()}`);
      }
      const body = await res.json() as { ok: boolean; data: { token: { id: string }; plaintext: string } };
      if (!body.ok) throw new Error(`mintDaemonToken: ok=false: ${JSON.stringify(body)}`);
      return { id: body.data.token.id, plaintext: body.data.plaintext };
    }

    /**
     * Calls a daemon endpoint with `Authorization: Bearer <plaintext>`.
     * Returns status + parsed body (never throws on non-2xx — lets tests assert explicit codes).
     */
    export async function callDaemonApi<T = unknown>(
      request: APIRequestContext,
      plaintext: string,
      method: 'GET' | 'POST' | 'DELETE',
      path: string,
      body?: unknown,
    ): Promise<{ status: number; body: T }> {
      const headers = { Authorization: `Bearer ${plaintext}` };
      let res;
      if (method === 'GET') {
        res = await request.get(path, { headers });
      } else if (method === 'POST') {
        res = await request.post(path, { headers, data: body ?? {} });
      } else {
        res = await request.delete(path, { headers });
      }
      const text = await res.text();
      let parsed: T;
      try { parsed = JSON.parse(text) as T; } catch { parsed = text as unknown as T; }
      return { status: res.status(), body: parsed };
    }

    /** Seed an issue + agent-task via Phase 17/18 REST so the daemon has something to claim. */
    export async function seedIssueWithTask(
      request: APIRequestContext,
      args: { runtimeId: string; agentId: string; title: string },
    ): Promise<{ issueId: string; taskId: string }> {
      // Create issue via POST /api/issues (Phase 17)
      const issueRes = await request.post('/api/issues', {
        data: { title: args.title, description: null, status: 'todo', priority: 'medium', assigneeId: args.agentId },
      });
      const issueBody = await issueRes.json() as { ok: boolean; data: { id: string } };
      if (!issueBody.ok) throw new Error(`seed issue failed: ${JSON.stringify(issueBody)}`);
      const issueId = issueBody.data.id;

      // Transition to todo → in_progress triggers enqueue (Phase 17 hook). Or just patch it.
      const patchRes = await request.patch(`/api/issues/${issueId}`, {
        data: { status: 'in_progress' },
      });
      if (patchRes.status() !== 200) throw new Error(`patch issue failed: ${patchRes.status()}`);

      // Read back the enqueued task id via GET /api/issues/:id (Phase 17 returns tasks[] in the issue payload).
      // Fallback: direct DB query is not available from Playwright — rely on Phase 17 response shape.
      const getRes = await request.get(`/api/issues/${issueId}`);
      const getBody = await getRes.json() as { ok: boolean; data: { tasks?: Array<{ id: string }> } };
      const taskId = getBody.data.tasks?.[0]?.id;
      if (!taskId) throw new Error(`no task found for issue ${issueId}`);

      return { issueId, taskId };
    }
    ```

    **Step B — write tests/e2e/daemon-rest.spec.ts** with:

    ```typescript
    import { test, expect, type APIRequestContext } from '@playwright/test';
    import { mintDaemonToken, callDaemonApi, seedIssueWithTask } from './fixtures/daemon-helpers';

    test.describe('Phase 19 — Daemon REST API & Auth', () => {
      // Helper: get a logged-in request context (CE mode has pass-through but test-cookie
      // flow is safer for CI). Use /api/auth/test-signup if present, else fall through.
      async function loggedInContext(browser): Promise<APIRequestContext> {
        const ctx = await browser.newContext();
        // Ping /api/users/me to force the CE pass-through to stamp a user row.
        await ctx.request.get('/api/users/me').catch(() => {});
        return ctx.request;
      }

      test('SC-1: register-happy — bearer works, cookie is 401', async ({ browser }) => {
        const user = await loggedInContext(browser);
        const { plaintext } = await mintDaemonToken(user, 'sc1-laptop');

        const { status, body } = await callDaemonApi<{ ok: boolean; data: { runtimes: unknown[] } }>(
          user, plaintext, 'POST', '/api/daemon/register',
          {
            daemonId: 'd1',
            deviceName: 'test-laptop',
            cliVersion: '0.1.0',
            launchedBy: 'pw-test',
            runtimes: [{ name: 'claude-cli', provider: 'claude', version: '1.0.0', status: 'online' }],
          },
        );
        expect(status).toBe(200);
        expect(body.ok).toBe(true);
        expect(Array.isArray(body.data.runtimes)).toBe(true);
        expect(body.data.runtimes.length).toBeGreaterThan(0);

        // Same endpoint, cookie auth (no bearer) — should be 401
        const cookieCtx = await browser.newContext();
        const res = await cookieCtx.request.post('/api/daemon/register', {
          data: { daemonId: 'd1', runtimes: [] },
        });
        expect(res.status()).toBe(401);
      });

      test('SC-2: privilege-confusion — cookie user with adt_* bearer is 401 on /api/agents', async ({ browser }) => {
        const user = await loggedInContext(browser);
        const { plaintext } = await mintDaemonToken(user, 'sc2-token');

        // Request /api/agents WITH the daemon bearer. The AUTH1 patch in requireAuth
        // must reject this even though the cookie session is valid.
        const res = await user.get('/api/agents', {
          headers: { Authorization: `Bearer ${plaintext}` },
        });
        expect(res.status()).toBe(401);
        const body = await res.json() as { ok: boolean; error: string };
        expect(body.ok).toBe(false);
        expect(body.error).toMatch(/daemon tokens not accepted on user routes/);
      });

      test('SC-3: rate-limit exemption — 400 daemon requests succeed, same count on /api/agents is 429', async ({ browser }) => {
        // Only meaningful when server is in production mode. Skip if NODE_ENV != production.
        test.skip(process.env.SERVER_NODE_ENV !== 'production', 'rate-limiter only on in production');

        const user = await loggedInContext(browser);
        const { plaintext } = await mintDaemonToken(user, 'sc3-token');

        // Register so the token has a daemon_id.
        await callDaemonApi(user, plaintext, 'POST', '/api/daemon/register', {
          daemonId: 'sc3-daemon',
          deviceName: 'sc3',
          cliVersion: '0',
          launchedBy: 'pw',
          runtimes: [{ name: 'x', provider: 'claude', version: '0', status: 'online' }],
        });

        // 400 heartbeats — exceeds the 300/15min static limiter but /api/daemon/* is exempt.
        let blocked = 0;
        for (let i = 0; i < 400; i++) {
          const { status } = await callDaemonApi(user, plaintext, 'POST', '/api/daemon/heartbeat', { runtimeIds: [] });
          if (status === 429) blocked++;
        }
        expect(blocked).toBe(0);

        // Same request count on /api/agents will eventually 429 (the user limiter is NOT exempt).
        // We do not assert the exact cutoff — just that at least one gets throttled.
        let throttled = false;
        for (let i = 0; i < 400 && !throttled; i++) {
          const res = await user.get('/api/agents');
          if (res.status() === 429) throttled = true;
        }
        expect(throttled).toBe(true);
      });

      test('SC-4: revocation-sla — revoked token rejected on next request within 1s', async ({ browser }) => {
        const user = await loggedInContext(browser);
        const { id, plaintext } = await mintDaemonToken(user, 'sc4-token');

        // Register.
        const regResult = await callDaemonApi<{ ok: boolean }>(
          user, plaintext, 'POST', '/api/daemon/register',
          {
            daemonId: 'sc4-daemon',
            deviceName: 'sc4',
            cliVersion: '0',
            launchedBy: 'pw',
            runtimes: [{ name: 'x', provider: 'claude', version: '0', status: 'online' }],
          },
        );
        expect(regResult.status).toBe(200);

        // Revoke.
        const revokeRes = await user.delete(`/api/daemon-tokens/${id}`);
        expect(revokeRes.status()).toBe(200);

        // Immediately re-call — measure elapsed.
        const t0 = Date.now();
        const { status } = await callDaemonApi(user, plaintext, 'POST', '/api/daemon/heartbeat', { runtimeIds: [] });
        const elapsed = Date.now() - t0;

        expect(status).toBe(401);
        expect(elapsed).toBeLessThan(1000);
      });

      test('SC-5: plaintext-once — create exposes plaintext, list does not', async ({ browser }) => {
        const user = await loggedInContext(browser);

        const createRes = await user.post('/api/daemon-tokens', { data: { name: 'sc5-token' } });
        expect(createRes.status()).toBe(200);
        const createBody = await createRes.json() as { ok: boolean; data: { token: Record<string, unknown>; plaintext: string } };
        expect(createBody.data.plaintext).toMatch(/^adt_.{32}$/);

        const listRes = await user.get('/api/daemon-tokens');
        expect(listRes.status()).toBe(200);
        const listBody = await listRes.json() as { ok: boolean; data: Array<Record<string, unknown>> };
        const serialized = JSON.stringify(listBody);
        // Plaintext MUST NOT appear in the list response anywhere.
        expect(serialized).not.toContain(createBody.data.plaintext);
        // Hash / token_hash / plaintext keys MUST NOT appear.
        expect(serialized).not.toMatch(/tokenHash|token_hash|plaintext/);
        // But the token's id must appear.
        const id = (createBody.data.token as { id: string }).id;
        expect(serialized).toContain(id);
      });

      test('full-story happy path — user→daemon→task→complete→revoke', async ({ browser }) => {
        const user = await loggedInContext(browser);
        const { id: tokenId, plaintext } = await mintDaemonToken(user, 'full-story');

        // Register daemon + one runtime.
        const regRes = await callDaemonApi<{ ok: boolean; data: { runtimes: Array<{ id: string }> } }>(
          user, plaintext, 'POST', '/api/daemon/register',
          {
            daemonId: 'fs-daemon',
            deviceName: 'fs',
            cliVersion: '0',
            launchedBy: 'pw',
            runtimes: [{ name: 'claude', provider: 'claude', version: '1.0', status: 'online' }],
          },
        );
        expect(regRes.status).toBe(200);
        const runtimeId = regRes.body.data.runtimes[0].id;

        // Create an agent bound to that runtime via the normal user API (Phase 17).
        const agentRes = await user.post('/api/agents', {
          data: { name: 'fs-agent', runtimeId, instructions: '', maxConcurrentTasks: 1 },
        });
        expect(agentRes.status()).toBe(200);
        const agentId = (await agentRes.json() as { data: { id: string } }).data.id;

        // Seed an issue + enqueue a task.
        const { taskId } = await seedIssueWithTask(user, { runtimeId, agentId, title: 'full-story task' });

        // Claim via daemon.
        const claimRes = await callDaemonApi<{ ok: boolean; data: { task: { id: string } | null } }>(
          user, plaintext, 'POST', `/api/daemon/runtimes/${runtimeId}/tasks/claim`, {},
        );
        expect(claimRes.status).toBe(200);
        expect(claimRes.body.data.task?.id).toBe(taskId);

        // Start.
        expect((await callDaemonApi(user, plaintext, 'POST', `/api/daemon/tasks/${taskId}/start`, {})).status).toBe(200);

        // Stream one message.
        expect((await callDaemonApi(user, plaintext, 'POST', `/api/daemon/tasks/${taskId}/messages`, {
          messages: [{ type: 'text', content: 'hello' }],
        })).status).toBe(200);

        // Complete.
        expect((await callDaemonApi(user, plaintext, 'POST', `/api/daemon/tasks/${taskId}/complete`, {
          result: { ok: true },
        })).status).toBe(200);

        // Status — should be 'completed'.
        const statusRes = await callDaemonApi<{ ok: boolean; data: { status: string } }>(
          user, plaintext, 'GET', `/api/daemon/tasks/${taskId}/status`,
        );
        expect(statusRes.body.data.status).toBe('completed');

        // Revoke and assert daemon is locked out.
        expect((await user.delete(`/api/daemon-tokens/${tokenId}`)).status()).toBe(200);
        const afterRevoke = await callDaemonApi(user, plaintext, 'POST', '/api/daemon/heartbeat', { runtimeIds: [runtimeId] });
        expect(afterRevoke.status).toBe(401);
      });
    });
    ```

  Public API:

  ```typescript
  export function appendTaskMessage(taskId: string, msg: PendingTaskMessage): void {
    const list = buffer.get(taskId) ?? [];
    list.push(msg);
    buffer.set(taskId, list);
    if (list.length >= BUFFER_SOFT_CAP) {
      // Fire-and-forget early flush to keep per-txn row count bounded.
      flushOne(taskId).catch((err) => {
        console.warn('[task-message-batcher] early flush failed for task', taskId, ':',
          err instanceof Error ? err.message : String(err));
      });
    }
  }

  export function startTaskMessageBatcher(): void {
    if (flushInterval) return;
    flushInterval = setInterval(() => {
      flushAll().catch((err) => console.warn('[task-message-batcher] flush failed:',
        err instanceof Error ? err.message : String(err)));
    }, BATCH_INTERVAL_MS);
    console.log('[task-message-batcher] started (500ms flush, 500-msg soft cap)');
  }

  export async function stopTaskMessageBatcher(): Promise<void> {
    if (flushInterval) {
      clearInterval(flushInterval);
      flushInterval = null;
    }
    // Final flush to avoid data loss on graceful shutdown.
    await flushAll();
    console.log('[task-message-batcher] stopped');
  }

  /** Test hook: flush one task (or all if omitted). Public so tests can force a flush without waiting 500ms. */
  export async function flushTaskMessages(taskId?: string): Promise<void> {
    if (taskId) return flushOne(taskId);
    return flushAll();
  }
  ```

  Internal helpers:

  ```typescript
  async function flushAll(): Promise<void> {
    const taskIds = [...buffer.keys()];
    for (const id of taskIds) {
      try { await flushOne(id); }
      catch (err) {
        console.warn('[task-message-batcher] flushOne failed for', id, ':',
          err instanceof Error ? err.message : String(err));
      }
    }
  }

  async function flushOne(taskId: string): Promise<void> {
    // Re-entrance guard: if a flush is already running for this task
    // (e.g. soft-cap overflow racing with timer tick), skip; the in-flight
    // flush will drain whatever is in the buffer.
    if (flushingTasks.has(taskId)) return;
    const pending = buffer.get(taskId);
    if (!pending || pending.length === 0) return;

    // Snapshot + clear the buffer BEFORE the DB call so new appends during
    // the flush accumulate for the next round (strict monotonicity per flush).
    const batch = pending.splice(0, pending.length);
    if (batch.length === 0) return;

    flushingTasks.add(taskId);
    try {
      const adapter = getAdapter();
      const toBroadcast: Array<{ workspaceId: string; issueId: string; seq: number; type: TaskMessageType; tool: string | null; content: string | null; input: unknown; output: unknown; }> = [];

      await db.transaction(async (trx) => {
        await trx.raw('BEGIN IMMEDIATE');
        const row = await trx('task_messages')
          .where({ task_id: taskId })
          .max({ m: 'seq' })
          .first();
        let next = Number(row?.m ?? 0);
        const inserts = batch.map((m) => {
          next += 1;
          toBroadcast.push({
            workspaceId: m.workspaceId, issueId: m.issueId, seq: next,
            type: m.type, tool: m.tool ?? null, content: m.content ?? null,
            input: m.input ?? null, output: m.output ?? null,
          });
          return {
            id: randomUUID(),
            task_id: taskId,
            seq: next,
            type: m.type,
            tool: m.tool ?? null,
            content: m.content ?? null,
            input: m.input === undefined ? null : adapter.jsonValue(m.input),
            output: m.output === undefined ? null : adapter.jsonValue(m.output),
            metadata: adapter.jsonValue(m.metadata ?? {}),
            created_at: db.fn.now(),
          };
        });
        // Single bulk insert keeps the write lock window minimal (SQ5 10ms budget).
        await trx('task_messages').insert(inserts);
      });

      // Broadcast AFTER commit — never inside the txn (PITFALLS §SQ5).
      for (const b of toBroadcast) {
        broadcast(b.workspaceId, {
          type: 'task:message', taskId, issueId: b.issueId,
          payload: {
            taskId, issueId: b.issueId, seq: b.seq, type: b.type,
            tool: b.tool, content: b.content, input: b.input, output: b.output,
          },
        });
      }
    } finally {
      flushingTasks.delete(taskId);
      // Drop the Map entry if buffer drained to keep Map small.
      if ((buffer.get(taskId)?.length ?? 0) === 0) buffer.delete(taskId);
    }
  }
  ```

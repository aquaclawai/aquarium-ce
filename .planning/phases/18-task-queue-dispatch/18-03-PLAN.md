  ```typescript
  import { db } from '../db/index.js';
  import { broadcast } from '../ws/index.js';

  /**
   * Task reaper — fails stale tasks in `agent_task_queue`.
   *
   * Scope (Phase 18, TASK-04):
   *   • dispatched rows older than DISPATCH_STALE_MS (5 min) -> failed
   *     (daemon claimed but never called startTask — crash between claim and start)
   *   • running rows older than RUNNING_STALE_MS (2.5 h) -> failed
   *     (daemon crashed mid-task, or child process deadlocked)
   *   • Runs every SWEEP_INTERVAL_MS (30 s). First sweep fires immediately on start.
   *   • Standalone module — clone of offline-sweeper.ts shape (16-RESEARCH §"Why NOT
   *     extend health-monitor.ts" applies: task and runtime lifecycles are separate).
   *   • Broadcasts `task:failed` per reaped row AFTER the UPDATE commits (never inside
   *     the write path — PITFALLS §SQ5).
   *
   * Race safety (PITFALLS §ST6):
   *   The reaper's UPDATE is guarded by both `status = '<stale-state>'` AND
   *   `<ts-column> < cutoff`. If a legitimate daemon transition (startTask /
   *   completeTask / failTask / cancelTask) commits first under pool=1, the
   *   reaper's WHERE no longer matches and affects 0 rows. Both orderings
   *   produce consistent terminal state (see 18-RESEARCH §"Race avoidance").
   */

  const DISPATCH_STALE_MS = 5 * 60_000;
  const RUNNING_STALE_MS  = 2.5 * 60 * 60_000;
  const SWEEP_INTERVAL_MS = 30_000;

  let sweepInterval: ReturnType<typeof setInterval> | null = null;

  export async function reapOnce(): Promise<{ dispatchedFailed: number; runningFailed: number }> {
    const now = Date.now();
    const dispatchCut = new Date(now - DISPATCH_STALE_MS).toISOString();
    const runningCut  = new Date(now - RUNNING_STALE_MS).toISOString();

    // SELECT IDs first so we can broadcast per-row after the batched UPDATE.
    const stuckDispatched = await db('agent_task_queue')
      .where('status', 'dispatched')
      .where('dispatched_at', '<', dispatchCut)
      .select('id', 'issue_id', 'workspace_id');
    const stuckRunning = await db('agent_task_queue')
      .where('status', 'running')
      .where('started_at', '<', runningCut)
      .select('id', 'issue_id', 'workspace_id');

    let dispatchedFailed = 0;
    let runningFailed = 0;

    if (stuckDispatched.length > 0) {
      dispatchedFailed = await db('agent_task_queue')
        .whereIn('id', stuckDispatched.map((r) => r.id as string))
        .andWhere('status', 'dispatched')     // ST6 race guard — daemon may have moved the row
        .update({
          status: 'failed',
          error: 'Reaper: dispatched > 5 min without start',
          completed_at: db.fn.now(),
          updated_at: db.fn.now(),
        });
    }
    if (stuckRunning.length > 0) {
      runningFailed = await db('agent_task_queue')
        .whereIn('id', stuckRunning.map((r) => r.id as string))
        .andWhere('status', 'running')        // ST6 race guard
        .update({
          status: 'failed',
          error: 'Reaper: running beyond configured timeout',
          completed_at: db.fn.now(),
          updated_at: db.fn.now(),
        });
    }

    // Broadcast AFTER commits — never inside a txn (PITFALLS §SQ5).
    // Since these are single-statement UPDATEs running outside any explicit
    // db.transaction, each UPDATE auto-commits before returning. Safe to broadcast here.
    for (const r of [...stuckDispatched, ...stuckRunning]) {
      broadcast(r.workspace_id as string, {
        type: 'task:failed',
        taskId: r.id as string,
        issueId: r.issue_id as string,
        payload: { taskId: r.id as string, issueId: r.issue_id as string },
      });
    }

    if (dispatchedFailed + runningFailed > 0) {
      console.log(`[task-reaper] failed ${dispatchedFailed} dispatched + ${runningFailed} running stuck task(s)`);
    }
    return { dispatchedFailed, runningFailed };
  }

  export function startTaskReaper(): void {
    if (sweepInterval) return;
    reapOnce().catch((err) => {
      console.warn('[task-reaper] initial sweep failed:', err instanceof Error ? err.message : String(err));
    });
    sweepInterval = setInterval(() => {
      reapOnce().catch((err) => {
        console.warn('[task-reaper] sweep failed:', err instanceof Error ? err.message : String(err));
      });
    }, SWEEP_INTERVAL_MS);
    console.log('[task-reaper] started (5min dispatched / 2.5h running thresholds, 30s sweep interval)');
  }

  export function stopTaskReaper(): void {
    if (sweepInterval) {
      clearInterval(sweepInterval);
      sweepInterval = null;
      console.log('[task-reaper] stopped');
    }
  }
  ```

| Boundary | Description |
|----------|-------------|
| reaper -> agent_task_queue | Internal process writes only. No external input; timing comes from setInterval. |
| reaper -> WebSocket subscribers | `task:failed` broadcasts fan out to workspace-subscribed clients; payload is taskId + issueId (no sensitive data). |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-18-13 | Tampering (ST6) | reaper fails a task that just legitimately transitioned (race between daemon startTask commit and reaper UPDATE) | mitigate | UPDATE uses `.whereIn('id', [...]).andWhere('status', 'dispatched')` — if daemon's UPDATE committed first (pool=1 serialisation), the status is now 'running' and reaper's WHERE no longer matches → zero rows affected. Test 5 (ST6 race safety) proves this empirically by calling startTask then reapOnce and asserting the row is still 'running'. |
| T-18-14 | Tampering (reaper reaps terminal tasks) | reaper UPDATE hits a row already in completed/failed/cancelled | mitigate | `.andWhere('status', 'dispatched')` / `.andWhere('status', 'running')` filters out terminal rows. Test 3 seeds terminal rows with ancient timestamps and verifies zero affected. |
| T-18-15 | DoS (long scans) | reaper table scan without index on stale tasks blocks DB | mitigate | Migration 007 already ships `idx_atq_claim` on `(runtime_id, status, priority, created_at)` and `idx_atq_issue_status` on `(issue_id, status)` — the reaper's `WHERE status='dispatched' AND dispatched_at < cutoff` partial-scans a small live subset. In CE single-user load, this is O(~10) rows per sweep. Accept: no custom index added in Phase 18; revisit only if production reaper p99 latency > 100ms. |
